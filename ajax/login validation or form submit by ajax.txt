    <script>
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();

            // Clear old errors
            document.getElementById('phone_error').innerText = '';
            document.getElementById('password_error').innerText = '';
            document.getElementById('loginError').style.display = 'none';

            let formData = new FormData(this);

            fetch(this.action, {
                    method: 'POST',
                    headers: {
                        'X-CSRF-TOKEN': '{{ csrf_token() }}',
                        'Accept': 'application/json'
                    },
                    body: formData
                })
                .then(async response => {
                    const data = await response.json();
                    if (!response.ok) throw data;
                    return data;
                })
                .then(data => {
                    if (data.status) {
                        window.location.href = data.redirect;
                    }
                })
                .catch(error => {
                    // Validation errors
                    if (error.errors) {
                        if (error.errors.phone) {
                            document.getElementById('phone_error').innerText = error.errors.phone[0];
                        }
                        if (error.errors.password) {
                            document.getElementById('password_error').innerText = error.errors.password[0];
                        }
                    }
                    // Auth error (wrong password)
                    else if (error.message) {
                        let err = document.getElementById('loginError');
                        err.style.display = 'block';
                        err.innerText = error.message;
                    }
                });
        });
    </script>

public function store(Request $request)
    {
        $data = $request->validate([
            'phone' => ['required', 'string'],
            'password' => ['required'],
        ]);

        $credential = ['phone' => $data['phone'], 'password' => $data['password']];
        if (!Auth::guard('customer')->attempt($credential)) {
            return response()->json([
                'status' => false,
                'message' => 'Your given credentials do not match our records'
            ], 401);
        } else {
            return response()->json([
                'status' => true,
                'redirect' => RouteServiceProvider::DASHBOARD,
            ]);
        }
    }
