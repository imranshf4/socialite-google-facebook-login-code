data() {
        return {
            categories: {},
            loading: true,
            search: '',
            currentPage: this.$route.query.page || 1,
            basePath: this.$store.state.image_base_link,
        }
    },

editCategoryByPosition(position) {
            const perPage = 50;
            const page = Math.ceil(position / perPage);
            const indexInPage = (position - 1) % perPage;

            axios.get('/api/list/category?page=' + page)
                .then(resp => {
                    if (resp.data.status) {
                        const category = resp.data.categories.data[indexInPage];
                        if (category) {
                            this.$router.push({
                                name: 'CategoryEdit',
                                params: { categoryId: category.id },
                                query: { page: page, pos: position }
                            });
                        }
                    }
                })
                .catch(error => {
                    this.$toastr.e('Category fetch failed!');
                });
        },


getCategories(page = this.currentPage) {
            this.loading = true;
            axios.get('/api/list/category?page=' + page)
                .then((resp) => {
                    if (resp.data.status) {
                        this.categories = resp.data.categories;
                        this.loading = false;
                        this.currentPage = page;

                        this.$router.replace({
                            path: this.$route.path,
                            query: { ...this.$route.query, page },
                        });

                        // Scroll to specific position after DOM updates
                        this.$nextTick(() => {
                            const pos = parseInt(this.$route.query.pos);
                            if (pos) {
                                const perPage = 50;
                                const indexInPage = (pos - 1) % perPage;
                                const tableRows = document.querySelectorAll('tbody tr');

                                if (tableRows[indexInPage]) {
                                    tableRows[indexInPage].scrollIntoView({
                                        behavior: 'smooth',
                                        block: 'center'
                                    });

                                    // highlight the row briefly
                                    tableRows[indexInPage].classList.add('highlight-row');
                                    setTimeout(() => {
                                        tableRows[indexInPage].classList.remove('highlight-row');
                                    }, 2000);
                                }
                            }
                        });
                    }
                })
                .catch(error => {
                    this.$toastr.e(error.response.data.message);
                    this.loading = false;
                });
        },

update() {
            this.form.post('/update/category/' + this.$route.params.categoryId, {
                transformRequest: [data => objectToFormData(data)],
            })
                .then(resp => {
                    if (resp.data.status) {
                        this.$toasted.show(resp.data.message, {
                            type: "success",
                            position: 'top-center',
                            duration: 4000
                        });

                        const page = this.$route.query.page || 1;
                        const pos = this.$route.query.pos;

                        // Go back to category list with both page and pos
                        this.$router.push({ name: 'category', query: { page, pos } });
                    }
                })
                .catch(error => {
                    this.$toastr.e(error.response.data.message);
                });
        },
